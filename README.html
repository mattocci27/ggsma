Of course, we can do the same thing without using show.sig.only argument.

dat_long2 <- dat_long %>%
  group_by(name) %>%
  nest() %>%
  mutate(p = map_dbl(data, ~ cor.test(.$y, .$value)$p.value)) %>%
  unnest(cols = c(data))

ggplot(dat_long2, aes(x = value, y = y, col = name)) +
  geom_point() +
  geom_sma(data = dat_long2 %>% filter(p < 0.05)) +
  geom_sma(data = dat_long2 %>% filter(p > 0.05),
           lty = 2, se = FALSE)
