Smooth lines for Standard Major Axis Regression

devtools::install_github("mattocci27/ggsma")
library(ggsma)
library(smatr)

data(leaflife)
leaf_low <- leaflife %>%
  filter(soilp == "low")

ggplot(leaflife, aes(lma, longev)) +
  geom_point() +
  geom_sma() +
  geom_smooth(method = "lm", col = "red")
